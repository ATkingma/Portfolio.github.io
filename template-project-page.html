Project Overview

The Network Security Scanner is a desktop application designed to help security professionals identify vulnerabilities in network infrastructure. Built with C# and integrating with popular Kali Linux tools, this application provides a user-friendly interface for conducting comprehensive network security assessments.

As part of my learning journey in cybersecurity, I developed this tool to automate common security tasks while providing detailed reporting capabilities. The project demonstrates my skills in both application development and security practices.

Technical Architecture

The Network Security Scanner follows a modular architecture with clearly separated components for scanning, analysis, and reporting:

Core Components

  • Scanner Engine: Manages network discovery and vulnerability scanning processes
  • Tool Integration Layer: Interfaces with Kali Linux security tools
  • Analysis Module: Processes scan results and identifies security issues
  • Reporting Engine: Generates comprehensive security reports
  • UI Framework: Provides an intuitive user interface built with WPF

        public async Task> ScanNetworkAsync(string subnetCIDR)
        {
            _discoveredHosts.Clear();
            
            // Parse CIDR notation to get network information
            var networkInfo = ParseCIDR(subnetCIDR);
            
            var scanTasks = new List();
            
            for (uint i = 1; i < networkInfo.HostCount - 1; i++)
            {
                var ipAddress = new IPAddress(networkInfo.NetworkAddress + i);
                scanTasks.Add(ScanHostAsync(ipAddress));
            }
            
            await Task.WhenAll(scanTasks);
            return _discoveredHosts;
        }
        
        private async Task ScanHostAsync(IPAddress hostAddress)
        {
            using (var ping = new Ping())
            {
                try
                {
                    var reply = await ping.SendPingAsync(hostAddress, _timeout);

                    {

using System;
using System.Net;
using System.Net.Sockets;
using System.Threading.Tasks;
using System.Collections.Generic;

namespace NetworkSecurityScanner.Core
{
    public class PortScanner
    {
        private readonly int _timeout;
        
        public class PortInfo
        {
            public int Port { get; set; }
            public bool IsOpen { get; set; }
            public string ServiceName { get; set; }
        }
        
        public PortScanner(int timeoutMs = 500)
        {
            _timeout = timeoutMs;
        }
        
        public async Task> ScanHostPortsAsync(IPAddress hostAddress, IEnumerable ports)
        {
            var results = new List();
            var scanTasks = new List>();
            
            foreach (var port in ports)
            {
                scanTasks.Add(ScanPortAsync(hostAddress, port));
            }
            
            var portResults = await Task.WhenAll(scanTasks);
            results.AddRange(portResults);
            
            return results;
        }
        
        private async Task ScanPortAsync(IPAddress hostAddress, int port)
        {
            var result = new PortInfo
            {
                Port = port,
                IsOpen = false,
                ServiceName = GetServiceName(port)
            };
            
            using (var tcpClient = new TcpClient())
            {
                try
                {
                    var connectTask = tcpClient.ConnectAsync(hostAddress, port);
                    if (await Task.WhenAny(connectTask, Task.Delay(_timeout)) == connectTask)
                    {
                        // The connection was successful within the timeout
                        result.IsOpen = true;
                    }
                }
                catch
                {
                    // Connection failed or timed out
                    result.IsOpen = false;
                }
            }
            
            return result;
        }
        
        private string GetServiceName(int port)
        {
            // Common port mappings
            Dictionary commonPorts = new Dictionary
            {
                { 21, "FTP" },
                { 22, "SSH" },
                { 23, "Telnet" },
                { 25, "SMTP" },
                { 53, "DNS" },
                { 80, "HTTP" },
                { 443, "HTTPS" },
                { 3306, "MySQL" },
                { 3389, "RDP" }
                // Additional ports could be added
            };
            
            if (commonPorts.TryGetValue(port, out string serviceName))
            {
                return serviceName;
            }
            
            return "Unknown";
        }
    }
}
