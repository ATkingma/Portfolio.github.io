Project Overview

This project is an interactive 2D pathfinding demo built in Unity and compiled to WebGL so you can run it live in the browser below. It implements the A* (A-star) algorithm to compute the shortest path between two points on a grid, navigating around obstacles in real time.

The demo lets you place walls, move the start and end nodes, and watch the algorithm explore and resolve the optimal path step by step. It showcases efficient heuristic-based search and grid-based movement commonly used in game AI.

Live Demo

The game runs directly in your browser via Unity WebGL. Click inside the canvas to interact.

How A* Works

The Algorithm: A* is a best-first search algorithm that finds the shortest path from a start node to a goal node on a weighted graph. It combines Dijkstra's guaranteed shortest path with a heuristic (usually Manhattan or Euclidean distance) to prioritize nodes that are closer to the goal, making it far more efficient.

f(n) = g(n) + h(n): Every node is scored by the sum of its actual cost from the start (g) and the estimated cost to the goal (h). Nodes with the lowest f score are explored first, ensuring the algorithm finds the optimal path while skipping unnecessary nodes.

Code Examples

A* Core Loop (C#) ⮞ 

using System.Collections.Generic;
using UnityEngine;

public class AStarPathfinder : MonoBehaviour
{
    public List<Node> FindPath(Node start, Node goal, Grid grid)
    {
        var openSet = new SortedSet<Node>(Comparer<Node>.Create((a, b) =>
            a.FCost != b.FCost ? a.FCost.CompareTo(b.FCost) : a.HCost.CompareTo(b.HCost)));

        var closedSet = new HashSet<Node>();

        start.GCost = 0;
        start.HCost = Heuristic(start, goal);
        openSet.Add(start);

        while (openSet.Count > 0)
        {
            var current = openSet.Min;
            openSet.Remove(current);

            if (current == goal)
                return ReconstructPath(goal);

            closedSet.Add(current);

            foreach (var neighbor in grid.GetNeighbors(current))
            {
                if (closedSet.Contains(neighbor) || !neighbor.Walkable) continue;

                float tentativeG = current.GCost + Vector2.Distance(current.Position, neighbor.Position);

                if (tentativeG < neighbor.GCost)
                {
                    neighbor.Parent = current;
                    neighbor.GCost = tentativeG;
                    neighbor.HCost = Heuristic(neighbor, goal);

                    if (!openSet.Contains(neighbor))
                        openSet.Add(neighbor);
                }
            }
        }

        return null; // No path found
    }

    private float Heuristic(Node a, Node b)
    {
        // Manhattan distance for grid-based movement
        return Mathf.Abs(a.GridX - b.GridX) + Mathf.Abs(a.GridY - b.GridY);
    }

    private List<Node> ReconstructPath(Node end)
    {
        var path = new List<Node>();
        var current = end;
        while (current != null)
        {
            path.Insert(0, current);
            current = current.Parent;
        }
        return path;
    }
}
Node Data Structure (C#) ⮞ 

using UnityEngine;

public class Node
{
    public bool Walkable;
    public Vector2 Position;
    public int GridX, GridY;

    public float GCost; // Cost from start
    public float HCost; // Heuristic cost to goal
    public float FCost => GCost + HCost;

    public Node Parent;

    public Node(bool walkable, Vector2 position, int gridX, int gridY)
    {
        Walkable = walkable;
        Position = position;
        GridX = gridX;
        GridY = gridY;
    }
}
